            limit = {

# AUTHORITARIAN
###############
country_event = {
    id = idrinthimprovedfactions.201
    title = "idrinthimprovedfactions.201.name"
    desc = "idrinthimprovedfactions.201.desc"
    picture = unity_symbol
    is_triggered_only = yes
    trigger = {
        has_event_chain = idrinthimprovedfactions_influence
        has_completed_event_chain_counter = {
            event_chain = idrinthimprovedfactions_influence
            counter = idrinthimprovedfactions_authoritarian
        }
    }
    immediate = {
        random_owned_planet = {
            random_list = {
                25 = {
                    create_army = {
                        name = "NAME_Faction_Guard"
                        owner = FROM
                        species = FROM
                        type = "assault_army"
                    }
                }
                75 = {}
            }
        }
        add_event_chain_counter = {
            event_chain = "idrinthimprovedfactions_influence"
            counter="idrinthimprovedfactions_authoritarian"
            amount = -1000
        }
    }
    option = {
        name = GREAT
    }
}
